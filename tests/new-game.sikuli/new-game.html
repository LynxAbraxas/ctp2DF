import shutil

def main():
    wait(, 100)
    click()
    click()
    if exists(, 100):
        if waitVanish(, 100): # control bar appears before progressbar vanishes
            file = capture(SCREEN.getBounds())
            if file:
                f= Finder(file) # http://doc.sikuli.org/finder.html
                f.find()
                if not f.hasNext():
                    print("Pattern not found in screen shot: " + file)
                    exit(30)
                shutil.move(file, 'ctp2new-game.png')
                exit(0)
        else:
            exit(20)
    else:
        exit(10)
    exit(99)

if __name__ == "__main__":
    main()
